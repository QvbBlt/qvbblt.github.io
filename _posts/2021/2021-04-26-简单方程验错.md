$$
\begin{cases}y-x=4 \\
(10y+x)-(10x+y)=27
\end{cases}
$$

```latex
\documentclass{article}
\begin{document}
  \begin{equation}
   %left right为闭合数组范围%
    \left\{\begin{array}{l}
    x-y=4 \\
    (10x+y)-(10y+x)=27
    \end{array}\right.
    \end{equation}
\end{document}
```
